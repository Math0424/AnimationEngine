BSL
Block Scripting Language

Object Delarations

	Subpart()
	Button(dummyName)
	Lever(dummyName, [x, y, z], time, angle)
	Emitter(dummyName)
	Emissive(emissiveID)
	# BasicIK(dummyNamme) - WIP, need SEUT support

Method format:
	object.method(xyz).delay(t)

ObjectMethods:
	
	Subpart.
		translate([x, y, z], time, lerp)
		rotate([x, y, z], angle, time, lerp)
		spin([x, y, z], speed, time)
		vibrate(time)
		setVisible(bool)
		reset()
		resetPos()
		setResetPos()
	
	#technically a subpart
	Button.
		enabled(bool)
		interactable(bool)

	#technically a subpart
	Lever.
		enabled(bool)
		interactable(bool)

	Emitter.
		playSound(p)
		playParticle(name, size, life, [x, y, z], r, g, b)
	
	Emissive.
		setColor(r, g, b, s)

		
	#still a WIP: !SEUT required!
